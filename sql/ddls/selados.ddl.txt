  `NUMTIPATO` char(3) NOT NULL,
  `IDAP` varchar(40) DEFAULT NULL,
  `CHAVE` varchar(30) DEFAULT NULL,
  `SELO` varchar(30) CHARACTER SET latin1 COLLATE latin1_general_cs DEFAULT NULL,
  `QRCODE` varchar(1000) DEFAULT NULL,
  `IMG` varchar(30000) DEFAULT NULL,
  `REGISTRO` varchar(30) DEFAULT NULL,
  `JSON` varchar(10000) DEFAULT NULL,
  `DATAENVIO` varchar(30) DEFAULT NULL,
  `DATARETORNO` varchar(30) DEFAULT NULL,
  `RETORNO` varchar(100) DEFAULT NULL,
  `STATUS` varchar(100) DEFAULT NULL,
  `USUARIO` varchar(100) DEFAULT NULL,
